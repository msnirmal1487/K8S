docker run ubuntu-sleeper 10

apiVersion: v1
kind: pod
metadata:
  name: ubuntu-sleeper-pod
spec:
  - name: ubuntu-sleeper
    image: ubuntu-sleeper
    args: 
      - 10 ## anything appended to the docker run command will go to the args of pod defeniton file as array


docker run --entrypoint sleep2.0 ubuntu-sleeper 10

apiVersion: v1
kind: pod
metadata:
  name: ubuntu-sleeper-pod
spec:
  - name: ubuntu-sleeper
    image: ubuntu-sleeper
    command: 
      - sleep2.0 ## to override the command in docker image (replaces command in ENTRYPOINT in DOCKERfile)
    args: 
      - 10 ## anything appended to the docker run command will go to the args of pod defeniton file as array (replaces the commandline CMD JSON array in dockerfile)

      OR
apiVersion: v1
kind: pod
metadata:
  name: ubuntu-sleeper-pod
spec:
  - name: ubuntu-sleeper
    image: ubuntu-sleeper
    command: 
      - sleep2.0 ## to override the command in docker image
        10

        OR
    
apiVersion: v1
kind: pod
metadata:
  name: ubuntu-sleeper-pod
spec:
  - name: ubuntu-sleeper
    image: ubuntu-sleeper
    command: ["sleep2.0", "20"] ## to override the command in docker image
